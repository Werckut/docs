            package net.werckut.testmod.setup;

            public class Registration {
                public static final DeferredRegister ITEMS = DeferredRegister.create(ForgeRegistries.ITEMS, TestMod.MOD_ID);

                public static void register() {
                    IEventBus modEventBus = FMLJavaModLoadingContext.get().getModEventBus();
                    ITEMS.register(modEventBus);

                    ModItems.register();
                }
            }
